cmake_minimum_required(VERSION 3.10)
#if the system is macos and the build type is release then set the osx architectures to arm64 and x86_64
if(APPLE)
  message(${CMAKE_BUILD_TYPE})
  if (CMAKE_BUILD_TYPE MATCHES "Release")
    message("Building a universal binary")
    set(CMAKE_OSX_ARCHITECTURES "arm64;x86_64")
  endif()
endif()
project(tooez LANGUAGES CXX)
file(GLOB SRC "src/*.cpp")
set(CMAKE_CXX_STANDARD 20)
add_executable(tooez ${SRC})
if(CMAKE_BUILD_TYPE MATCHES "Debug")
  target_compile_options(tooez PRIVATE -Wno-everything)
  target_compile_options(tooez PRIVATE -g)
endif()
target_include_directories(tooez PUBLIC ${CMAKE_CURRENT_SOURCE_DIR} /usr/local/include)
target_link_libraries(tooez PUBLIC)
if(NOT DEFINED CMAKE_BUILD_TYPE)
  set(CMAKE_OSX_ARCHITECTURES arm64;x86_64)
  set(CMAKE_BUILD_TYPE Release)
endif()
